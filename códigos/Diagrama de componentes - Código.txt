@startuml
left to right direction

package "Sistema de Gestão das Olimpíadas (SGO)" {
    
    node "Frontend" {
        [Interface de Usuário] as UI
    }
    
    node "Backend (Regras de Negócio)" {
        ' Definição dos Módulos
        component "Módulo de Competições" as ModComp
        component "Módulo de Inscrições" as ModInsc
        component "Módulo de Alocação" as ModAloc
        component "Módulo de Relatórios" as ModRel
        
        ' Interfaces Fornecidas (O que cada módulo faz/oferece para o sistema)
        interface "ICompeticao\n(Cadastrar, Consultar)" as IComp
        interface "IInscricao\n(Inscrever Atleta, Validar País)" as IInsc
        interface "IAlocacao\n(Alocar Local, Checar Conflito)" as IAloc
        interface "IResultado\n(Registrar Pódio, Gerar Quadro)" as IRel
        
        ' Ligando os módulos às suas interfaces fornecidas
        ModComp -up- IComp
        ModInsc -up- IInsc
        ModAloc -up- IAloc
        ModRel -up- IRel
        
        ' Dependências internas (Interfaces Requeridas - Como interagem)
        ' Inscrições, Alocações e Relatórios precisam consultar as Competições
        ModInsc ..> IComp : "Requer dados da Competição"
        ModAloc ..> IComp : "Vincula Local à Competição"
        ModRel ..> IComp : "Registra pódio da Competição"
    }
    
    database "Armazenamento" {
        [Base de Dados SGO] as BD
    }
}

' A Interface de Usuário consome as interfaces (APIs) disponibilizadas pelo backend
UI ..> IComp : <<use>>
UI ..> IInsc : <<use>>
UI ..> IAloc : <<use>>
UI ..> IRel : <<use>>

' Todos os módulos acessam a base de dados
ModComp ..> BD : <<use>>
ModInsc ..> BD : <<use>>
ModAloc ..> BD : <<use>>
ModRel ..> BD : <<use>>

@enduml
