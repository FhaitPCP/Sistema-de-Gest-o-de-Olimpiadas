package "GestaoDeParticipantes" {
    class País {
        - nome: String
        - medalhasOuro: Integer
        - medalhasPrata: Integer
        - medalhasBronze: Integer
        + atualizarQuadroMedalhas(): void
    }

    class Atleta {
        - nome: String
        - passaporte: String
    }
}

package "GestaoDeEventos" {
    class Local {
        - nome: String
        - capacidade: Integer
        + verificarDisponibilidade(data: Date, horario: Time): Boolean
    }

    class Competição {
        - nomeModalidade: String
        - data: Date
        - horario: Time
        + validarAlocacao(): Boolean
    }
    
    class Inscricao <<Association>> {
        - dataInscricao: Date
        + validarPaisUnicoPorModalidade(): Boolean
    }
}

package "GestaoDeResultados" {
    class Resultado {
        + registrarPodio(ouro: Atleta, prata: Atleta, bronze: Atleta): void
        + gerarRelatorioMedalhas(): void
    }
}

' Relacionamentos
Atleta "1" -- "*" Inscricao
Competição "1" -- "*" Inscricao

Competição "*" -- "1" Local : "Ocorre em"
Competição "1" -- "0..1" Resultado : "Possui"

Resultado --> "1" Atleta : "1º Lugar (Ouro)"
Resultado --> "1" Atleta : "2º Lugar (Prata)"
